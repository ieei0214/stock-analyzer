<project_specification>
  <project_name>Stock Analyzer</project_name>

  <overview>
    A Python-based multi-agent stock analysis system with a web dashboard interface. The system uses two specialized agents: a Data Collector that fetches comprehensive stock data (price history, financials, news, insider trading, analyst ratings) and stores it in SQLite, and an LLM-powered Investment Analyst that evaluates the data to provide Buy/Sell/Hold recommendations with detailed reasoning. Users access the system through a localhost web interface to manage watchlists, trigger analyses, view reports with embedded charts, and track historical recommendation performance.
  </overview>

      <charts>Chart.js or Plotly.js for interactive visualizations</charts>
    </frontend>

      <runtime>Python 3.10+</runtime>

  <core_features>
    <agent_system>
      - Multi-agent architecture with two specialized agents
      - Agent 1: Data Collector (fetches and stores stock data)
      - Agent 2: Investment Analyst (LLM-powered analysis and recommendations)
      - Asynchronous agent execution for non-blocking operations
      - Agent communication via shared SQLite database
      - Error recovery and retry logic for agent failures
    </agent_system>

    <data_collection>
      - Fetch historical price data (1 year) via yfinance
      - Retrieve key financial metrics: P/E ratio, EPS, Revenue Growth
      - Fetch dividend yield, debt-to-equity ratio, profit margins
      - Get analyst ratings and price targets
      - Collect insider trading activity
      - Search and store recent news headlines with sentiment indicators
      - Calculate technical indicators: 50-day MA, 200-day MA, RSI, MACD
      - Store all raw data in SQLite with timestamps
      - Data validation and error handling for missing/incomplete data
    </data_collection>

    <llm_analysis>
      - Configurable LLM provider (OpenAI or Gemini via .env)
      - System prompt for investment analysis with structured output
      - Fundamental analysis: P/E vs industry average, revenue trends, debt levels
      - Technical analysis: price trends, moving average crossovers, RSI/MACD signals
      - Sentiment analysis: news headline sentiment aggregation
      - Generate Buy/Sell/Hold recommendation with confidence level
      - Provide bullet-point reasoning for recommendation
      - Support for conservative vs aggressive analysis styles
      - Fallback error handling for API failures or rate limits
    </llm_analysis>

    <web_dashboard>
      - Home/Dashboard page with watchlist overview
      - Display current price, % change, last recommendation for each watchlist stock
      - Analyze Stock page with ticker input and configuration options
      - Stock Detail page showing full analysis report with embedded charts
      - History/Performance page for tracking past recommendations
      - Settings page for API key management and preferences
      - Responsive design for desktop and tablet viewing
      - Real-time loading indicators for ongoing analyses
      - Toast notifications for completed analyses and errors
    </web_dashboard>

    <watchlist_management>
      - Add stocks to watchlist by ticker symbol
      - Remove stocks from watchlist
      - View all watchlist stocks on dashboard
      - "Analyze All" button to trigger batch analysis of all watchlist stocks
      - Background processing for batch analyses with progress tracking
      - Persistent storage of watchlist in SQLite
      - Display last analysis date for each watchlist stock
    </watchlist_management>

    <report_generation>
      - Generate detailed markdown reports for each analysis
      - Embed chart images directly in markdown files
      - Charts: price history with MAs, volume, RSI, MACD, revenue/EPS trends
      - Include all financial metrics in structured format
      - Show news headlines with sentiment scores
      - Display insider trading activity summary
      - Highlight Buy/Sell/Hold recommendation prominently
      - Save reports to organized directory structure: reports/TICKER/YYYY-MM-DD_analysis.md
      - Link to reports from web UI for easy access
    </report_generation>

    <historical_tracking>
      - Store all past analyses with timestamps in database
      - Manual "Check Performance" trigger to evaluate past recommendations
      - Calculate gain/loss from recommendation date to current/selected date
      - Show performance metrics: 1 week, 1 month, 3 months, 6 months
      - Display success rate statistics (% of Buy/Sell recommendations that were profitable)
      - Historical view with filterable date ranges
      - Compare recommendation price vs current price
      - Visual indicators for profitable vs unprofitable recommendations
    </historical_tracking>

    <caching_system>
      - Configurable cache duration in settings (1 hour, 6 hours, 12 hours, 1 day, until market close)
      - Cache fetched stock data to reduce API calls and improve performance
      - Timestamp-based cache validation
      - Manual "Refresh Data" option to bypass cache
      - Cache indicator in UI showing data freshness
      - Automatic cache cleanup for expired data
    </caching_system>

    <settings_configuration>
      - Edit API keys for OpenAI and Gemini
      - Select active LLM provider (OpenAI or Gemini)
      - Choose default analysis style (Conservative or Aggressive)
      - Set cache duration preference
      - Test API connection from settings page
      - Save settings to .env file or database
      - Reset to default settings option
    </settings_configuration>

    <error_handling>
      - Graceful handling of invalid ticker symbols with user-friendly messages
      - LLM API failure recovery with retry logic and fallback messages
      - Network error handling with timeout configurations
      - Missing data handling (e.g., no P/E ratio, limited history for new IPOs)
      - Rate limit detection and user notification
      - yfinance API errors with helpful suggestions
      - Display error details in UI without exposing sensitive information
      - Log errors to file for debugging
    </error_handling>

    <background_processing>
      - Task queue for batch "Analyze All" operations
      - Progress tracking UI for multi-stock analyses
      - Parallel processing with rate limiting to avoid API throttling
      - Notification system when background tasks complete
      - Cancel/abort option for long-running analyses
      - Queue status visibility in UI
    </background_processing>
  </core_features>

  <database_schema>
    <tables>
      <stocks>
        - ticker (TEXT, PRIMARY KEY)
        - company_name (TEXT)
        - sector (TEXT)
        - industry (TEXT)
        - created_at (TIMESTAMP)
        - last_updated (TIMESTAMP)
      </stocks>

      <watchlist>
        - id (INTEGER, PRIMARY KEY AUTOINCREMENT)
        - ticker (TEXT, FOREIGN KEY to stocks.ticker)
        - added_at (TIMESTAMP)
      </watchlist>

      <stock_data>
        - id (INTEGER, PRIMARY KEY AUTOINCREMENT)
        - ticker (TEXT, FOREIGN KEY to stocks.ticker)
        - data_type (TEXT: 'price_history', 'fundamentals', 'technicals', 'news')
        - data_json (TEXT: JSON blob of actual data)
        - fetched_at (TIMESTAMP)
        - cache_expires_at (TIMESTAMP)
      </stock_data>

      <analyses>
        - id (INTEGER, PRIMARY KEY AUTOINCREMENT)
        - ticker (TEXT, FOREIGN KEY to stocks.ticker)
        - analysis_date (TIMESTAMP)
        - recommendation (TEXT: 'Buy', 'Sell', 'Hold')
        - confidence_level (TEXT: 'High', 'Medium', 'Low')
        - reasoning (TEXT: bullet points)
        - analysis_style (TEXT: 'Conservative', 'Aggressive')
        - price_at_analysis (REAL)
        - report_path (TEXT: path to markdown file)
        - llm_provider (TEXT: 'openai', 'gemini')
        - llm_model (TEXT: model name used)
      </analyses>

      <performance_checks>
        - id (INTEGER, PRIMARY KEY AUTOINCREMENT)
        - analysis_id (INTEGER, FOREIGN KEY to analyses.id)
        - check_date (TIMESTAMP)
        - price_at_check (REAL)
        - gain_loss_percent (REAL)
        - time_period_days (INTEGER)
        - outcome (TEXT: 'Profitable', 'Loss', 'Neutral')
      </performance_checks>

      <news_data>
        - id (INTEGER, PRIMARY KEY AUTOINCREMENT)
        - ticker (TEXT, FOREIGN KEY to stocks.ticker)
        - headline (TEXT)
        - source (TEXT)
        - published_date (TIMESTAMP)
        - sentiment (TEXT: 'Positive', 'Negative', 'Neutral')
        - url (TEXT)
        - fetched_at (TIMESTAMP)
      </news_data>

      <settings>
        - key (TEXT, PRIMARY KEY)
        - value (TEXT)
        - updated_at (TIMESTAMP)
      </settings>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <watchlist_endpoints>
      - GET /api/watchlist (Get all watchlist stocks)
      - POST /api/watchlist (Add stock to watchlist)
      - DELETE /api/watchlist/{ticker} (Remove stock from watchlist)
      - POST /api/watchlist/analyze-all (Trigger batch analysis)
    </watchlist_endpoints>

    <analysis_endpoints>
      - POST /api/analyze/{ticker} (Trigger single stock analysis)
      - GET /api/analysis/{ticker} (Get latest analysis for ticker)
      - GET /api/analysis/{ticker}/history (Get all past analyses)
      - GET /api/analysis/{analysis_id}/report (Get markdown report)
    </analysis_endpoints>

    <stock_data_endpoints>
      - GET /api/stock/{ticker} (Get stock overview data)
      - GET /api/stock/{ticker}/data (Get cached stock data)
      - POST /api/stock/{ticker}/refresh (Force refresh cached data)
      - GET /api/stock/{ticker}/charts (Get chart images)
    </stock_data_endpoints>

    <performance_endpoints>
      - POST /api/performance/check/{analysis_id} (Trigger performance check)
      - GET /api/performance/stats (Get overall performance statistics)
      - GET /api/performance/history (Get all performance checks)
    </performance_endpoints>

    <settings_endpoints>
      - GET /api/settings (Get all settings)
      - PUT /api/settings (Update settings)
      - POST /api/settings/test-api (Test LLM API connection)
      - POST /api/settings/reset (Reset to defaults)
    </settings_endpoints>

    <background_task_endpoints>
      - GET /api/tasks/{task_id}/status (Get task status)
      - GET /api/tasks/active (Get all active tasks)
      - POST /api/tasks/{task_id}/cancel (Cancel task)
    </background_task_endpoints>
  </api_endpoints_summary>
